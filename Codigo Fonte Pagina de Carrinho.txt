function applyNewLayout(row) {
    // Salva o estado original do elemento tr
    const originalRowHTML = row.innerHTML;

    // Seleciona todos os tds da linha
    const tds = row.querySelectorAll('td:not(.hidden-phone)'); // Exclui tds com a classe 'hidden-phone'

    // Cria as novas divs
    const firstTdDiv = document.createElement('div');
    firstTdDiv.className = 'first-td-div';
    firstTdDiv.innerHTML = tds[0].innerHTML;

    const otherTdsDiv = document.createElement('div');
    otherTdsDiv.className = 'other-tds-div';
    otherTdsDiv.style.display = 'flex';
    otherTdsDiv.style.flexDirection = 'column';
    otherTdsDiv.style.justifyContent = 'space-between';
    otherTdsDiv.style.width = '100%';

    const groupedDiv = document.createElement('div');
    groupedDiv.className = 'grouped-tds-div';
    groupedDiv.style.display = 'flex';
    groupedDiv.style.gap = '5px';
    groupedDiv.style.alignItems = 'center';
    groupedDiv.style.justifyContent = 'space-around';

    // Move o conteúdo dos tds para as novas divs
    for (let i = 1; i < tds.length; i++) {
        const div = document.createElement('div');
        div.className = `td-${i + 1}`;
        div.innerHTML = tds[i].innerHTML;
        (i >= 2 ? groupedDiv : otherTdsDiv).appendChild(div);
    }

    otherTdsDiv.appendChild(groupedDiv);

    // Cria um novo td para adicionar as divs
    const newTd = document.createElement('td');
    newTd.style.display = 'flex';
    newTd.append(firstTdDiv, otherTdsDiv);

    // Aplica o novo layout à linha
    row.style.display = 'flex';
    row.style.flexDirection = 'column';
    row.appendChild(newTd);

    // Remove os tds antigos
    tds.forEach(td => td.remove());

    // Retorna a função de restauração para o estado original
    return function restoreOriginalLayout() {
        row.innerHTML = originalRowHTML;
        row.style.display = ''; // Remove a estilização aplicada anteriormente
    };
}

// Função para alternar entre o novo layout e o original com base no tamanho da tela
function handleResponsiveLayout() {
    const rows = document.querySelectorAll('tr[data-produto-id]');
    const restoreFunctions = [];

    function applyOrRestoreLayout() {
        if (window.innerWidth < 768) {
            rows.forEach((row, index) => {
                if (!restoreFunctions[index]) {
                    restoreFunctions[index] = applyNewLayout(row);
                }
            });
        } else {
            restoreFunctions.forEach((restoreOriginal, index) => {
                if (restoreOriginal) {
                    restoreOriginal();
                    restoreFunctions[index] = null;
                }
            });
        }
    }

    // Aplica ou restaura o layout ao carregar a página
    applyOrRestoreLayout();

    // Adiciona um listener para aplicar/restaurar o layout ao redimensionar a tela
    window.addEventListener('resize', applyOrRestoreLayout);
    window.addEventListener('load', applyOrRestoreLayout);
}

// Chama a função de layout responsivo ao carregar a página
handleResponsiveLayout();

    // Copia conteúdo (ícone + texto) do botão original para o flutuante

    // Funções para mostrar e esconder o flutuante

          floatBtn.classList.add('visivel');
        });

        floatBtn.classList.remove('visivel');
        floatBtn.addEventListener('transitionend', function handler(e) {
          if (e.propertyName === 'opacity') {
            floatBtn.style.display = 'none';
            floatBtn.removeEventListener('transitionend', handler);
          }
        });
      }
    }

    // Verifica se o original saiu totalmente da viewport

      original = pegaOriginal();

      }

      // Função para definir cookie
function setCookie(cname, cvalue, exdays) {
    var d = new Date();
    d.setTime(d.getTime() + (exdays*24*60*60*1000));
    var expires = "expires="+ d.toUTCString();
    document.cookie = cname + "=" + cvalue + ";" + expires + ";path=/";
}

// Função para ler cookie
function getCookie(cname) {
    var name = cname + "=";
    var decodedCookie = decodeURIComponent(document.cookie);
    var ca = decodedCookie.split(';');
    for(var i = 0; i <ca.length; i++) {
        var c = ca[i];
        while (c.charAt(0) == ' ') {
            c = c.substring(1);
        }
        if (c.indexOf(name) === 0) {
            return c.substring(name.length, c.length);
        }
    }
    return "";
}

(function(){
  function ajustarLauncher(){
    const path = window.location.pathname.toLowerCase();
    const shouldHide = path.includes('carrinho');

    const lf = document.querySelector('.widget-visible > iframe:first-of-type');
    if (lf) {
      if (shouldHide) {
        lf.style.display = 'none';
      } else {
        lf.style.display        = '';
        lf.style.bottom         = '14px';
        lf.style.width          = '40px';
        lf.style.height         = '40px';
        lf.style.transform      = 'scale(0.8) translateX(-20px)';
        lf.style.transformOrigin= 'bottom right';
      }
    }

    const icon = document.querySelector('.tawk-min-chat-icon');
    if (icon) {
      if (shouldHide) {
        icon.style.display = 'none';
      } else {
        icon.style.display        = '';
        icon.style.width          = '24px';
        icon.style.height         = '24px';
        icon.style.transform      = 'scale(0.8) translateX(-20px)';
        icon.style.transformOrigin= 'bottom right';
      }
    }
  }

  // Ganchos oficiais do Tawk.to
  window.Tawk_API = window.Tawk_API || {};
  ['onLoad', 'onChatMaximized', 'onChatMinimized'].forEach(evt => {
    window.Tawk_API[evt] = () => {
      if (evt === 'onChatMinimized') {
        setTimeout(ajustarLauncher, 50);
      } else {
        ajustarLauncher();
      }
    };
  });

  // Fallback robusto: observa recriações do iframe/ícone
  const observer = new MutationObserver(ajustarLauncher);
  observer.observe(document.body, { childList: true, subtree: true });
  setTimeout(() => observer.disconnect(), 10000);

  // (Opcional) seu setInterval original
  let tries = 0;
  const iv = setInterval(() => {
    ajustarLauncher();
    if (++tries > 10) clearInterval(iv);
  }, 500);
})();
